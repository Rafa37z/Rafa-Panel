local function updateAllColors()
    if not UI_Elements.mainFrame then return end
    
    -- Elementos principais
    UI_Elements.mainFrame.BackgroundColor3 = Colors.Background
    UI_Elements.colorOverlay.BackgroundColor3 = Colors.Primary
    UI_Elements.shadow.ImageColor3 = Colors.Primary
    UI_Elements.topBar.BackgroundColor3 = Colors.Surface
    UI_Elements.title.TextColor3 = Colors.Primary
    UI_Elements.versionLabel.TextColor3 = Colors.Secondary
    UI_Elements.closeButton.ImageColor3 = Colors.TextSecondary
    
    -- Abas
    for i, btn in ipairs(UI_Elements.tabButtons) do
        if btn then
            btn.BackgroundColor3 = Colors.Surface
            btn.TextColor3 = (i == currentTab) and Colors.Primary or Colors.TextSecondary
        end
    end
    
    -- Scrollbars
    for i, content in ipairs(UI_Elements.tabContents) do
        if content and content.Frame then
            content.Frame.ScrollBarImageColor3 = Colors.ScrollBar
        end
    end
    
    for _, scroll in ipairs(UI_Elements.scrollBars) do
        if scroll and scroll.Parent and scroll ~= UI_Elements.themeScroll then
            scroll.ScrollBarImageColor3 = Colors.ScrollBar
        end
    end
    
    -- Conteúdo das abas
    for _, content in ipairs(UI_Elements.tabContents) do
        if content and content.Frame then
            for _, descendant in ipairs(content.Frame:GetDescendants()) do
                
                -- IGNORA botões de tema
                if descendant:IsA("TextButton") and descendant.Parent == UI_Elements.themeScroll then
                    continue
                end
                if descendant:IsA("Frame") and descendant.Parent and descendant.Parent:IsA("TextButton") and descendant.Parent.Parent == UI_Elements.themeScroll then
                    continue
                end
                if descendant:IsA("TextLabel") and descendant.Parent and descendant.Parent:IsA("TextButton") and descendant.Parent.Parent == UI_Elements.themeScroll then
                    continue
                end
                
                if descendant:IsA("Frame") then
                    if descendant.Name == "PerfilCard" or descendant.Name == "InfoCard" or 
                       descendant.Name == "ThemeCard" or descendant.Name == "RestartCard" then
                        descendant.BackgroundColor3 = Colors.Surface
                    end
                    
                    if descendant.Parent and descendant.Parent:IsA("ScrollingFrame") then
                        if not descendant:FindFirstChild("IsSlider") then
                            descendant.BackgroundColor3 = Colors.Surface
                        end
                    end
                    
                    if descendant:FindFirstChild("TextLabel") and descendant.Size.Y.Offset <= 25 then
                        local text = descendant:FindFirstChild("TextLabel")
                        if text and #text.Text <= 3 then
                            descendant.BackgroundColor3 = Colors.Primary
                            descendant.BackgroundTransparency = 0.7
                        end
                    end
                    
                    if descendant.Parent and descendant.Parent:IsA("Frame") and descendant.Size.Y.Offset <= 6 then
                        if descendant:FindFirstChild("Frame") then
                            descendant.BackgroundColor3 = Colors.SliderBackground
                            UI_Elements.sliders[descendant] = true
                        end
                    end
                    
                    if descendant:IsA("Frame") and descendant.Parent and descendant.Parent:IsA("Frame") then
                        if descendant.Parent.Parent and descendant.Parent.Parent:IsA("Frame") then
                            if descendant.Parent.Size.Y.Offset <= 6 then
                                descendant.BackgroundColor3 = Colors.SliderFill
                            end
                        end
                    end
                end
                
                if descendant:IsA("TextLabel") or descendant:IsA("TextButton") then
                    if descendant:IsA("TextButton") and (descendant.Text == "ON" or descendant.Text == "OFF") then
                        if descendant.Text == "ON" then
                            descendant.BackgroundColor3 = Colors.Primary
                        else
                            descendant.BackgroundColor3 = Colors.Danger
                        end
                        descendant.TextColor3 = Colors.Text
                    
                    elseif descendant:IsA("TextButton") and descendant.Text == "VIEW" then
                        if viewTarget and descendant.Parent and descendant.Parent:FindFirstChild("TextLabel") then
                            local nameLabel = descendant.Parent:FindFirstChild("TextLabel")
                            if nameLabel and nameLabel.Text == viewTarget.Name then
                                descendant.BackgroundColor3 = Colors.ViewActiveColor
                            else
                                descendant.BackgroundColor3 = Colors.ViewColor
                            end
                        else
                            descendant.BackgroundColor3 = Colors.ViewColor
                        end
                        descendant.TextColor3 = Colors.Text
                    
                    elseif descendant:IsA("TextButton") and descendant.Text == "TP" then
                        descendant.BackgroundColor3 = Colors.Primary
                        descendant.TextColor3 = Colors.Text
                    
                    elseif descendant:IsA("TextButton") and (descendant.Text == "★" or descendant.Text == "☆") then
                        -- Botão de favorito - mantém cor dourada
                        if descendant.Text == "★" then
                            descendant.TextColor3 = Color3.fromRGB(255, 215, 0)
                        else
                            descendant.TextColor3 = Colors.TextSecondary
                        end
                    
                    elseif descendant:IsA("TextButton") and descendant.Text == "TOME" then
                        -- Botão TOME - cor especial
                        descendant.BackgroundColor3 = Color3.fromRGB(255, 128, 0) -- Laranja
                        descendant.TextColor3 = Colors.Text
                    
                    elseif descendant:IsA("TextButton") then
                        if descendant.Text == "SALVAR CONFIGURAÇÕES" or descendant.Text == "REINICIAR" then
                            descendant.BackgroundColor3 = Colors.Primary
                        elseif descendant.Text == "RESETAR CONFIGURAÇÕES" then
                            descendant.BackgroundColor3 = Color3.fromRGB(255, 80, 80)
                        end
                        descendant.TextColor3 = Colors.Text
                    
                    elseif descendant:IsA("TextLabel") then
                        if descendant.Text == "TEMAS DISPONÍVEIS" or descendant.Text == "REINICIAR PAINEL" then
                            descendant.TextColor3 = Colors.Primary
                        
                        elseif descendant.Parent and descendant.Parent:FindFirstChild("Frame") then
                            local line = descendant.Parent:FindFirstChild("Frame")
                            if line and line.Size.Y.Offset <= 1 then
                                descendant.BackgroundColor3 = Colors.Surface
                                descendant.BackgroundTransparency = 0.3
                                descendant.TextColor3 = Colors.Primary
                                if line then line.BackgroundColor3 = Colors.Primary end
                            end
                        
                        elseif descendant.Parent and descendant.Parent:IsA("Frame") then
                            if descendant.Parent.Size.Y.Offset >= 70 then
                                if descendant.Position.Y.Offset <= 15 then
                                    descendant.TextColor3 = Colors.Text
                                elseif descendant.Position.Y.Offset >= 30 then
                                    descendant.TextColor3 = Colors.TextSecondary
                                end
                            end
                        
                        elseif descendant.Text:match("%d+%.?%d*") or descendant.Text:match("%d+") then
                            descendant.TextColor3 = Colors.ValueText
                        
                        else
                            descendant.TextColor3 = Colors.TextSecondary
                        end
                    end
                end
                
                if descendant:IsA("Frame") and descendant.Size.Y.Offset <= 1 and descendant.BackgroundTransparency < 1 then
                    if descendant.Parent and descendant.Parent:FindFirstChild("TextLabel") then
                        descendant.BackgroundColor3 = Colors.Primary
                    end
                end
            end
        end
    end
    
    -- Value labels
    for _, label in ipairs(UI_Elements.valueLabels) do
        if label and label.Parent then
            label.TextColor3 = Colors.ValueText
        end
    end
    
    -- Search frame
    if UI_Elements.searchFrame then
        UI_Elements.searchFrame.BackgroundColor3 = Colors.Surface
        local icon = UI_Elements.searchFrame:FindFirstChild("TextLabel")
        if icon and icon.Text == "🔍" then
            icon.TextColor3 = Colors.Primary
        end
        local box = UI_Elements.searchFrame:FindFirstChild("TextBox")
        if box then
            box.PlaceholderColor3 = Colors.TextSecondary
            box.TextColor3 = Colors.Text
        end
    end
    
    -- Player frames (já são atualizados separadamente)
    
    -- Info cards
    for _, card in ipairs(UI_Elements.infoCards) do
        if card and card.Parent then
            card.BackgroundColor3 = Colors.Surface
        end
    end
    
    -- FPS Label
    if UI_Elements.fpsLabel and UI_Elements.fpsLabel.Parent then
        UI_Elements.fpsLabel.BackgroundColor3 = Colors.Surface
        local fps = tonumber(UI_Elements.fpsLabel.Text:match("%d+")) or 60
        if fps >= 60 then
            UI_Elements.fpsLabel.TextColor3 = Colors.HealthGreen
        elseif fps >= 30 then
            UI_Elements.fpsLabel.TextColor3 = Colors.HealthYellow
        else
            UI_Elements.fpsLabel.TextColor3 = Colors.HealthRed
        end
    end
    
    -- ESP
    if States.esp then
        for _, highlight in pairs(espHighlights) do
            if highlight then
                highlight.FillColor = Colors.Primary
                highlight.OutlineColor = Colors.Primary
            end
        end
    end
end

end

local Performance = {
    PartCache = {},
    ConnectionList = {}
}

function Performance:CacheParts(character)
    if not character then return {} end
    if self.PartCache[character] then return self.PartCache[character] end
    
    local parts = {}
    for _, descendant in ipairs(character:GetDescendants()) do
        if descendant:IsA("BasePart") then
            table.insert(parts, descendant)
        end
    end
    
    self.PartCache[character] = parts
    character.AncestryChanged:Connect(function()
        self.PartCache[character] = nil
    end)
    
    return parts
end

function Performance:TrackConnection(conn)
    table.insert(self.ConnectionList, conn)
    return conn
end

local ConfigManager = {
    FileName = "Rafa37_Config.json",
    BackupName = "Rafa37_Config_Backup.json"
}

function ConfigManager:Validate(data)
    if type(data) ~= "table" then return false end
    for k, v in pairs(Config) do
        if data[k] == nil then
            data[k] = v
        elseif type(data[k]) ~= type(v) then
            data[k] = v
        end
    end
    return true
end

function ConfigManager:Save()
    if not Config.SaveSettings then 
        NotificationSystem:Show("Config", "Salvamento desativado", "info")
        return false
    end
    
    local success, json = pcall(HttpService.JSONEncode, HttpService, Config)
    if success and writefile then
        writefile(self.FileName, json)
        writefile(self.BackupName, json)
        NotificationSystem:Show("Config", "Configurações salvas!", "success")
        return true
    end
    NotificationSystem:Show("Config", "Erro ao salvar", "error")
    return false
end

function ConfigManager:Load()
    if not Config.AutoLoad then return end
    
    local data = nil
    pcall(function()
        if isfile and isfile(self.FileName) then
            data = HttpService:JSONDecode(readfile(self.FileName))
        end
    end)
    
    if not data then
        pcall(function()
            if isfile and isfile(self.BackupName) then
                data = HttpService:JSONDecode(readfile(self.BackupName))
            end
        end)
    end
    
    if data and self:Validate(data) then
        for k, v in pairs(data) do 
            Config[k] = v 
        end
        return true
    end
    return false
end

local NotificationSystem = {
    queue = {},
    container = nil,
    activeNotifications = 0,
    lastNotification = {title = "", message = "", time = 0}
}

function NotificationSystem:Init(parent)
    self.container = Instance.new("Frame")
    self.container.Name = "Notifications"
    self.container.Size = UDim2.new(0, 300, 1, -40)
    self.container.Position = UDim2.new(1, -320, 0, 20)
    self.container.BackgroundTransparency = 1
    self.container.ZIndex = 200
    self.container.Parent = parent
    
    -- Layout para organizar as notificações
    local layout = Instance.new("UIListLayout")
    layout.Padding = UDim.new(0, 8)
    layout.HorizontalAlignment = Enum.HorizontalAlignment.Right
    layout.VerticalAlignment = Enum.VerticalAlignment.Bottom
    layout.SortOrder = Enum.SortOrder.LayoutOrder
    layout.Parent = self.container
end

function NotificationSystem:Show(title, message, nType)
    nType = nType or "info"
    
    -- Verifica se a mesma notificação já foi mostrada nos últimos 2 segundos
    local now = tick()
    if self.lastNotification.title == title and self.lastNotification.message == message and now - self.lastNotification.time < 2 then
        return -- Ignora duplicatas
    end
    
    -- Atualiza a última notificação
    self.lastNotification = {
        title = title,
        message = message,
        time = now
    }
    
    -- Cores baseadas no tipo
    local colors = {
        info = Colors.Primary,
        success = Colors.Success,
        warning = Color3.fromRGB(255, 170, 0),
        error = Color3.fromRGB(255, 50, 50)
    }
    
    -- Cria a notificação imediatamente
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, 0, 0, 60)
    frame.BackgroundColor3 = Colors.Surface
    frame.BackgroundTransparency = 0.1
    frame.BorderSizePixel = 0
    frame.ZIndex = 201
    frame.Parent = self.container
    frame.Visible = false
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = frame
    
    local bar = Instance.new("Frame")
    bar.Size = UDim2.new(0, 4, 1, 0)
    bar.BackgroundColor3 = colors[nType]
    bar.BorderSizePixel = 0
    bar.ZIndex = 202
    bar.Parent = frame
    
    local titleLabel = Instance.new("TextLabel")
    titleLabel.Size = UDim2.new(1, -20, 0, 25)
    titleLabel.Position = UDim2.new(0, 15, 0, 8)
    titleLabel.BackgroundTransparency = 1
    titleLabel.Text = title
    titleLabel.TextColor3 = colors[nType]
    titleLabel.TextXAlignment = Enum.TextXAlignment.Left
    titleLabel.Font = Enum.Font.GothamBold
    titleLabel.TextSize = 14
    titleLabel.ZIndex = 202
    titleLabel.Parent = frame

--[[
    ========== SISTEMA REVERSO CORRIGIDO - FINAL ==========
]]
local Reverso = {
    history = {},           -- Tabela de posições e rotações
    maxHistory = 300,       -- 15 segundos a 0.05s = 300 posições
    recording = false,
    connection = nil,
    rewindConnection = nil,
    lastData = nil,
    currentIndex = nil
}

function Reverso:startRecording()
    if self.recording then return end
    self.recording = true
    self.history = {}
    self.lastData = nil
    
    -- Conexão para registrar posições a cada 0.05s (inclusive no ar)
    self.connection = RunService.Heartbeat:Connect(function()
        if not States.reversoEnabled or not self.recording or States.reversoActive then
            return
        end
        
        local character = player.Character
        if not character then return end
        
        local rootPart = character:FindFirstChild("HumanoidRootPart")
        if not rootPart then return end
        
        -- Salva POSIÇÃO E ROTAÇÃO completas (funciona no ar também)
        local currentData = {
            cframe = rootPart.CFrame,  -- Salva o CFrame completo (posição + rotação)
            time = tick()
        }
        
        -- Evita registrar dados muito próximos (otimização)
        if self.lastData then
            local distance = (currentData.cframe.Position - self.lastData.cframe.Position).Magnitude
            if distance < 0.1 then
                return
            end
        end
        
        table.insert(self.history, currentData)
        self.lastData = currentData
        
        -- Mantém apenas os últimos maxHistory registros
        if #self.history > self.maxHistory then
            table.remove(self.history, 1)
        end
    end)
end

function Reverso:stopRecording()
    self.recording = false
    if self.connection then
        self.connection:Disconnect()
        self.connection = nil
    end
end

function Reverso:startRewind()
    if #self.history == 0 or States.reversoActive then return end
    
    States.reversoActive = true
    self.currentIndex = #self.history
    
    -- Desativa outras funções que podem conflitar
    if States.fly then toggleFly(false) end
    if States.tomeActive then toggleTome(tomeTarget) end
    
    -- Prepara o personagem
    local character = player.Character
    if character then
        local humanoid = character:FindFirstChild("Humanoid")
        if humanoid then
            humanoid.PlatformStand = true
            humanoid.AutoRotate = false
        end
    end
    
    -- Pega o tempo inicial
    local startTime = tick()
    local startIndex = self.currentIndex
    
    -- Loop de rewind (VELOCIDADE ORIGINAL)
    self.rewindConnection = RunService.Heartbeat:Connect(function()
        if not States.reversoActive or not States.reversoEnabled then
            self:stopRewind()
            return
        end
        
        if self.currentIndex and self.currentIndex > 0 then
            local character = player.Character
            if not character then
                self:stopRewind()
                return
            end
            
            local rootPart = character:FindFirstChild("HumanoidRootPart")
            if not rootPart then return end
            
            local currentData = self.history[self.currentIndex]
            if not currentData then
                self.currentIndex = self.currentIndex - 1
                return
            end
            
            -- Vai para a posição
            rootPart.CFrame = currentData.cframe
            
            -- Calcula o tempo que deve esperar até o próximo frame
            if self.currentIndex > 1 then
                local nextData = self.history[self.currentIndex - 1]
                if nextData then
                    -- Tempo entre os dois pontos no histórico
                    local timeDiff = currentData.time - nextData.time
                    -- Espera exatamente o mesmo tempo
                    task.wait(timeDiff)
                end
            end
            
            self.currentIndex = self.currentIndex - 1
        else
            self:stopRewind()
        end
    end)
    
    NotificationSystem:Show("Reverso", "Rebobinando (velocidade original)", "info")
end

function Reverso:stopRewind()
    if not States.reversoActive then return end
    
    States.reversoActive = false
    
    if self.rewindConnection then
        self.rewindConnection:Disconnect()
        self.rewindConnection = nil
    end
    
    -- Restaura o personagem
    local character = player.Character
    if character then
        local humanoid = character:FindFirstChild("Humanoid")
        if humanoid then
            humanoid.PlatformStand = false
            humanoid.AutoRotate = true
        end
    end
    
    NotificationSystem:Show("Reverso", "Parou", "info")
end

    ========== SARRADA ==========
]]
local function toggleTome(targetPlayer)
    -- Se já está seguindo o mesmo jogador, desativa
    if States.tomeActive and tomeTarget == targetPlayer then
        -- Para o movimento
        if tomeConnection then
            tomeConnection:Disconnect()
            tomeConnection = nil
        end
        
        -- Restaura o personagem ao normal
        local character = player.Character
        if character then
            local humanoid = character:FindFirstChild("Humanoid")
            if humanoid then
                humanoid.AutoRotate = true
                humanoid.PlatformStand = false
            end
        end
        
        States.tomeActive = false
        tomeTarget = nil
        NotificationSystem:Show("SARRADA", "Parou", "info")
        
        -- Atualiza a lista para mudar a cor do botão
        if updatePlayerList then updatePlayerList() end
        return
    end
    
    -- Se está seguindo outro jogador, para primeiro
    if tomeConnection then
        tomeConnection:Disconnect()
        tomeConnection = nil
    end
    
    -- Verifica se o alvo é válido
    if not targetPlayer or targetPlayer == player then
        NotificationSystem:Show("SARRADA", "Selecione um jogador válido", "error")
        return
    end
    
    local targetChar = targetPlayer.Character
    if not targetChar then
        NotificationSystem:Show("SARRADA", "Jogador sem personagem", "error")
        return
    end
    
    local targetRoot = targetChar:FindFirstChild("HumanoidRootPart")
    if not targetRoot then
        NotificationSystem:Show("SARRADA", "Não foi possível", "error")
        return
    end
    
    -- Desativa fly se estiver ativo
    if States.fly then
        toggleFly(false)
    end
    
    -- Ativa o modo SARRADA
    States.tomeActive = true
    tomeTarget = targetPlayer
    
    -- Distâncias do movimento
    local baseDistance = 1.8
    local distanciaMinima = 1.2
    local time = 0
    
    tomeConnection = RunService.Heartbeat:Connect(function(dt)
        if not States.tomeActive or not tomeTarget then return end
        
        local currentTarget = tomeTarget
        local targetChar = currentTarget.Character
        if not targetChar then
            toggleTome(currentTarget)
            return
        end
        
        local targetRoot = targetChar:FindFirstChild("HumanoidRootPart")
        if not targetRoot then return end
        
        local localChar = player.Character
        if not localChar then return end
        
        local localRoot = localChar:FindFirstChild("HumanoidRootPart")
        if not localRoot then return end
        
        local lookVector = targetRoot.CFrame.LookVector
        
        -- Velocidade exponencial
        local velocidadeBase = 2 + (Config.TomeSpeed * Config.TomeSpeed * 1.5)
        time = time + dt * velocidadeBase
        
        local amplitude = 0.3 + (Config.TomeSpeed * 0.25)
        amplitude = math.min(amplitude, 1.2)
        
        local movimento = math.sin(time) * amplitude
        local distancia = baseDistance + movimento
        distancia = math.max(distancia, distanciaMinima)
        
        local targetPos = targetRoot.Position - lookVector * distancia
        targetPos = Vector3.new(targetPos.X, targetRoot.Position.Y, targetPos.Z)
        
        localRoot.CFrame = CFrame.new(targetPos, targetRoot.Position)
    end)
    
    NotificationSystem:Show("SARRADA", "Ativada", "success")
    if updatePlayerList then updatePlayerList() end
end

local function toggleAimbot(state)
    States.aimbot = state
    if Toggles.aimbot then Toggles.aimbot.setState(state) end
    
    safeDisconnect(Connections.aimbot)
    
    if state then
        Connections.aimbot = Performance:TrackConnection(RunService.Heartbeat:Connect(function()
            if not States.aimbot then return end
            
            if Config.StealthMode then task.wait(math.random(1, 5) / 100) end
            
            local bestDist = Config.AimbotFOV
            local bestTarget = nil
            local bestPos = nil
            
            for _, otherPlayer in ipairs(Players:GetPlayers()) do
                if otherPlayer ~= player and otherPlayer.Character and otherPlayer.Character:FindFirstChild("Head") then
                    local head = otherPlayer.Character.Head
                    local headPos = head.Position
                    local screenPos, onScreen = camera:WorldToViewportPoint(headPos)
                    
                    if onScreen then
                        local dist = (Vector2.new(mouse.X, mouse.Y) - Vector2.new(screenPos.X, screenPos.Y)).Magnitude
                        if dist < bestDist then
                            bestDist = dist
                            bestTarget = otherPlayer
                            bestPos = headPos
                        end
                    end
                end
            end
            
            if bestTarget and bestPos then
                if Config.HumanLikeBehavior then
                    local error = Vector3.new(
                        (math.random() - 0.5) * 0.1,
                        (math.random() - 0.5) * 0.05,
                        (math.random() - 0.5) * 0.1
                    )
                    bestPos = bestPos + error
                    
                    local currentLook = camera.CFrame.LookVector
                    local targetLook = (bestPos - camera.CFrame.Position).Unit
                    local smoothLook = currentLook:Lerp(targetLook, Config.AimbotSmoothness)
                    camera.CFrame = CFrame.new(camera.CFrame.Position, camera.CFrame.Position + smoothLook)
                else
                    camera.CFrame = CFrame.new(camera.CFrame.Position, bestPos)
                end
            end
        end))
        
        NotificationSystem:Show("Aim Bot", "Ativado (K)", "success")
    else
        NotificationSystem:Show("Aim Bot", "Desativado (K)", "info")
    end
end

--[[
    ========== ANTI AFK CORRIGIDO - VERSÃO FINAL ==========
]]
local antiAfkConnection = nil
local antiAfkActive = false

local function toggleAntiAfk(state)
    States.antiAfk = state
    antiAfkActive = state
    if Toggles.antiAfk then
        Toggles.antiAfk.setState(state)
    end
    
    -- Remove conexão anterior se existir
    if antiAfkConnection then
        antiAfkConnection:Disconnect()
        antiAfkConnection = nil
    end
    
    if state then
        local lastPos = nil
        local lastMove = tick()
        local moveCounter = 0
        
        -- Verifica a cada 1 segundo (mais eficiente que Heartbeat)
        antiAfkConnection = RunService.Stepped:Connect(function()
            -- Verifica se o anti-afk ainda está ativo
            if not antiAfkActive then
                if antiAfkConnection then
                    antiAfkConnection:Disconnect()
                    antiAfkConnection = nil
                end
                return
            end
            
            -- Verifica se o personagem existe
            local character = player.Character
            if not character then return end
            
            local humanoid = character:FindFirstChild("Humanoid")
            local rootPart = character:FindFirstChild("HumanoidRootPart")
            
            if not humanoid or not rootPart then return end
            
            -- Verifica se o personagem está em modo TOME, FLY ou REVERSO
            if States.tomeActive or States.fly or States.reversoActive then
                lastPos = rootPart.Position
                lastMove = tick()
                return
            end
            
            local currentPos = rootPart.Position
            
            -- Se é a primeira vez, apenas registra a posição
            if not lastPos then
                lastPos = currentPos
                lastMove = tick()
                return
            end
            
            -- Verifica se o personagem está parado (movimento muito pequeno)
            local distanceMoved = (currentPos - lastPos).Magnitude
            
            -- Se moveu, atualiza
            if distanceMoved > 0.5 then
                lastPos = currentPos
                lastMove = tick()
                return
            end
            
            -- Se ficou parado por mais de 15 segundos
            if tick() - lastMove > 15 then
                -- Move um pouco para não ser kickado
                moveCounter = moveCounter + 1
                
                -- Força um movimento usando CFrame (mais confiável que MoveTo)
                local moveDistance = 2
                
                -- Alterna entre direções
                if moveCounter % 4 == 0 then
                    rootPart.CFrame = rootPart.CFrame + Vector3.new(moveDistance, 0, 0)
                elseif moveCounter % 4 == 1 then
                    rootPart.CFrame = rootPart.CFrame + Vector3.new(-moveDistance, 0, 0)
                elseif moveCounter % 4 == 2 then
                    rootPart.CFrame = rootPart.CFrame + Vector3.new(0, 0, moveDistance)
                else
                    rootPart.CFrame = rootPart.CFrame + Vector3.new(0, 0, -moveDistance)
                end
                
                -- Pequena pausa para o movimento
                task.wait(0.1)
                
                -- Volta para posição original (opcional, mas parece mais natural)
                rootPart.CFrame = rootPart.CFrame - Vector3.new(moveDistance, 0, 0)
                
                -- Atualiza timer e posição
                lastPos = rootPart.Position
                lastMove = tick()
            end
        end)
        
        NotificationSystem:Show("Anti AFK", "Ativado", "success")
    end
end

local function teleportToPlayer(targetPlayer)
    if not targetPlayer or targetPlayer == player then
        NotificationSystem:Show("TP", "Jogador inválido", "error")
        return
    end
    
    local character = player.Character
    local targetChar = targetPlayer.Character
    
    if not character or not targetChar then
        NotificationSystem:Show("TP", "Jogador sem personagem", "error")
        return
    end
    
    local root = character:FindFirstChild("HumanoidRootPart")
    local targetRoot = targetChar:FindFirstChild("HumanoidRootPart")
    
    if not root or not targetRoot then
        NotificationSystem:Show("TP", "Não foi possível teleportar", "error")
        return
    end
    
    root.CFrame = targetRoot.CFrame * CFrame.new(0, 3, 0)
    
    local effect = Instance.new("Part")
    effect.Size = Vector3.new(3, 3, 3)
    effect.Shape = Enum.PartType.Ball
    effect.BrickColor = BrickColor.new(Colors.Primary)
    effect.Material = Enum.Material.Neon
    effect.CanCollide = false
    effect.Anchored = true
    effect.Position = root.Position
    effect.Transparency = 0.3
    effect.Parent = workspace
    
    task.spawn(function()
        for i = 1, 15 do
            effect.Transparency = effect.Transparency + 0.05
            effect.Size = effect.Size + Vector3.new(0.3, 0.3, 0.3)
            task.wait(0.02)
        end
        effect:Destroy()
    end)
    
    NotificationSystem:Show("TP", "Teleportado para " .. targetPlayer.Name, "success")
end

--[[
    ========== VIEW MODE ==========
]]
local function endViewMode()
    safeDisconnect(Connections.view)
    safeDisconnect(Connections.viewMouse)
    safeDisconnect(Connections.viewWheel)
    safeDisconnect(Connections.viewESC)
    
    camera.CameraType = Enum.CameraType.Custom
    
    for _, btn in ipairs(viewButtons) do
        if btn then btn.BackgroundColor3 = Colors.ViewColor end
    end
    
    viewActive = false
    viewTarget = nil
    
    NotificationSystem:Show("View", "Visualização encerrada", "info")
end

local function viewPlayer(targetPlayer)
    if viewActive and viewTarget == targetPlayer then
        endViewMode()
        return
    end
    
    if viewActive then endViewMode() end
    
    if not targetPlayer or targetPlayer == player then
        NotificationSystem:Show("View", "Jogador inválido", "error")
        return
    end
    
    local targetChar = targetPlayer.Character
    if not targetChar then
        NotificationSystem:Show("View", "Jogador sem personagem", "error")
        return
    end
    
    local targetRoot = targetChar:FindFirstChild("HumanoidRootPart")
    if not targetRoot then
        NotificationSystem:Show("View", "Não foi possível visualizar", "error")
        return
    end
    
    viewTarget = targetPlayer
    viewActive = true
    
    for _, btn in ipairs(viewButtons) do
        if btn and btn.Parent then
            local nameLabel = btn.Parent:FindFirstChild("TextLabel")
            if nameLabel and nameLabel.Text == targetPlayer.Name then
                btn.BackgroundColor3 = Colors.ViewActiveColor
            else
                btn.BackgroundColor3 = Colors.ViewColor
            end
        end
    end
    
    camera.CameraType = Enum.CameraType.Scriptable
    
    local camDist = 10
    local camHeight = 3
    local camAngle = 0
    local mouseSens = 0.005
    local zoomSpeed = 2
    
    local function updateCam()
        if not viewActive then return end
        if not targetPlayer or not targetPlayer.Character then
            endViewMode()
            return
        end
        
        local root = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
        if not root then
            endViewMode()
            return
        end
        
        local offset = Vector3.new(
            math.sin(camAngle) * camDist,
            camHeight,
            math.cos(camAngle) * camDist
        )
        
        camera.CFrame = CFrame.new(root.Position + offset, root.Position + Vector3.new(0, 2, 0))
    end
    
    Connections.viewMouse = Performance:TrackConnection(UserInputService.InputChanged:Connect(function(input)
        if viewActive and input.UserInputType == Enum.UserInputType.MouseMovement then
            local delta = input.Delta
            camAngle = camAngle + delta.X * mouseSens
            camHeight = math.clamp(camHeight - delta.Y * mouseSens * 10, 1, 8)
            updateCam()
        end
    end))
    
    Connections.viewWheel = Performance:TrackConnection(UserInputService.InputChanged:Connect(function(input)
        if viewActive and input.UserInputType == Enum.UserInputType.MouseWheel then
            camDist = math.clamp(camDist - input.Position.Z * zoomSpeed, 3, 30)
            updateCam()
        end
    end))
    
    Connections.viewESC = Performance:TrackConnection(UserInputService.InputBegan:Connect(function(input)
        if input.KeyCode == Enum.KeyCode.Escape and viewActive then
            endViewMode()
        end
    end))
    
    Connections.view = Performance:TrackConnection(RunService.Heartbeat:Connect(updateCam))
    
    NotificationSystem:Show("View", "Visualizando " .. targetPlayer.Name, "info")
end

local ChatCommandsActive = false
local ChatConnection = nil

local function toggleChatCommands(state)
    Config.ChatCommands = state
    if Toggles.chatCommands then Toggles.chatCommands.setState(state) end
    
    -- Remove conexão anterior
    if ChatConnection then
        ChatConnection:Disconnect()
        ChatConnection = nil
    end
    
    if state then
        ChatConnection = player.Chatted:Connect(function(msg)
            if msg:sub(1,1) ~= "/" then return end
            
            local args = string.split(msg:sub(2), " ")
            local cmd = args[1]:lower()
            
            if cmd == "fly" then
                toggleFly(not States.fly)
            elseif cmd == "esp" then
                toggleESP(not States.esp)
            elseif cmd == "aim" or cmd == "aimbot" then
                toggleAimbot(not States.aimbot)
            elseif cmd == "speed" then
                toggleSpeed(not States.speed)
            elseif cmd == "jump" then
                toggleJump(not States.jump)
            elseif cmd == "noclip" then
                toggleNoclip(not States.noclip)
            elseif cmd == "tp" and args[2] then
                for _, p in ipairs(Players:GetPlayers()) do
                    if p.Name:lower():find(args[2]:lower()) then
                        teleportToPlayer(p)
                        break
                    end
                end
            elseif cmd == "theme" and args[2] then
                for i, theme in ipairs(Themes) do
                    if theme.Name:lower():find(args[2]:lower()) then
                        applyTheme(i)
                        break
                    end
                end
            elseif cmd == "save" then
                ConfigManager:Save()
            elseif cmd == "reset" then
                resetAllSettings()
            elseif cmd == "light" then
                toggleLightMode(not Config.LightMode)
            elseif cmd == "tome" and args[2] then
                for _, p in ipairs(Players:GetPlayers()) do
                    if p.Name:lower():find(args[2]:lower()) then
                        toggleTome(p)
                        break
                    end
                end
            elseif cmd == "reverso" then
                Reverso:toggle(not States.reversoEnabled)
            elseif cmd == "help" then
                NotificationSystem:Show("Comandos", "/fly /esp /aim /tp /theme /save /reset /light /tome /reverso", "info", 5)
            end
        end)
        
        ChatCommandsActive = true
        NotificationSystem:Show("Chat", "Comandos ativados! Use /help", "success")
    else
        ChatCommandsActive = false
        NotificationSystem:Show("Chat", "Comandos desativados", "info")
    end
end

end

-- Função para atualizar lista de jogadores (com favoritos e botão TOME)
local function updatePlayerList()
    for _, child in ipairs(playerListFrame:GetChildren()) do
        if child:IsA("Frame") and child.Name ~= "UIListLayout" then
            child:Destroy()
        end
    end
    
    viewButtons = {}
    UI_Elements.playerFrames = {}
    local searchText = searchBox.Text:lower()
    
    -- Separa jogadores em favoritos e não favoritos
    local favoritePlayers = {}
    local normalPlayers = {}
    
    for _, otherPlayer in ipairs(Players:GetPlayers()) do
        if otherPlayer ~= player then
            if Favorites:IsFavorite(otherPlayer.Name) then
                table.insert(favoritePlayers, otherPlayer)
            else
                table.insert(normalPlayers, otherPlayer)
            end
        end
    end
    
    -- Função para criar o frame do jogador
    local function createPlayerFrame(targetPlayer, isFavorite)
        local playerName = targetPlayer.Name:lower()
        if searchText ~= "" and not playerName:find(searchText) then
            return
        end
        
        local pFrame = Instance.new("Frame")
        pFrame.Size = UDim2.new(1, 0, 0, 50)
        pFrame.BackgroundColor3 = Colors.Surface
        pFrame.BackgroundTransparency = 0.1
        pFrame.BorderSizePixel = 0
        pFrame.Parent = playerListFrame
        table.insert(UI_Elements.playerFrames, pFrame)
        
        local pCorner = Instance.new("UICorner")
        pCorner.CornerRadius = UDim.new(0, 6)
        pCorner.Parent = pFrame
        
        -- Nome do jogador
        local nameLabel = Instance.new("TextLabel")
        nameLabel.Size = UDim2.new(1, -240, 1, 0)  -- Mais espaço para 3 botões
        nameLabel.Position = UDim2.new(0, 10, 0, 0)
        nameLabel.BackgroundTransparency = 1
        nameLabel.Text = targetPlayer.Name
        nameLabel.TextColor3 = Colors.Text
        nameLabel.TextXAlignment = Enum.TextXAlignment.Left
        nameLabel.Font = Enum.Font.GothamBold
        nameLabel.TextSize = 14
        nameLabel.ZIndex = 18
        nameLabel.Parent = pFrame
        
        -- Botão de favorito (estrela)
        local favBtn = Instance.new("TextButton")
        favBtn.Size = UDim2.new(0, 30, 0, 30)
        favBtn.Position = UDim2.new(1, -230, 0.5, -15)
        favBtn.BackgroundTransparency = 1
        favBtn.Text = isFavorite and "★" or "☆"
        favBtn.TextColor3 = isFavorite and Color3.fromRGB(255, 215, 0) or Colors.TextSecondary
        favBtn.Font = Enum.Font.GothamBold
        favBtn.TextSize = 20
        favBtn.ZIndex = 19
        favBtn.Parent = pFrame
        
        favBtn.MouseButton1Click:Connect(function()
            Favorites:Toggle(targetPlayer.Name)
            -- Atualiza apenas este botão
            if Favorites:IsFavorite(targetPlayer.Name) then
                favBtn.Text = "★"
                favBtn.TextColor3 = Color3.fromRGB(255, 215, 0)
            else
                favBtn.Text = "☆"
                favBtn.TextColor3 = Colors.TextSecondary
            end
            -- Recria a lista para reordenar
            updatePlayerList()
        end)
        
        -- Botão VIEW
        local viewBtn = Instance.new("TextButton")
        viewBtn.Size = UDim2.new(0, 50, 0, 30)
        viewBtn.Position = UDim2.new(1, -175, 0.5, -15)
        viewBtn.BackgroundColor3 = Colors.ViewColor
        viewBtn.Text = "VIEW"
        viewBtn.TextColor3 = Colors.Text
        viewBtn.Font = Enum.Font.GothamBold
        viewBtn.TextSize = 10
        viewBtn.ZIndex = 19
        viewBtn.Parent = pFrame
        
        if viewTarget == targetPlayer and viewActive then
            viewBtn.BackgroundColor3 = Colors.ViewActiveColor
        end
        
        local viewCorner = Instance.new("UICorner")
        viewCorner.CornerRadius = UDim.new(0, 6)
        viewCorner.Parent = viewBtn
        
        viewBtn.MouseButton1Click:Connect(function()
            viewPlayer(targetPlayer)
        end)
        
        table.insert(viewButtons, viewBtn)
        
        -- Botão TOME
        local tomeBtn = Instance.new("TextButton")
        tomeBtn.Size = UDim2.new(0, 50, 0, 30)
        tomeBtn.Position = UDim2.new(1, -115, 0.5, -15)
        
        if States.tomeActive and tomeTarget == targetPlayer then
            tomeBtn.BackgroundColor3 = Color3.fromRGB(255, 0, 0) -- Vermelho quando ativo
        else
            tomeBtn.BackgroundColor3 = Color3.fromRGB(255, 128, 0) -- Laranja
        end
        
        tomeBtn.Text = "TOME"
        tomeBtn.TextColor3 = Colors.Text
        tomeBtn.Font = Enum.Font.GothamBold
        tomeBtn.TextSize = 10
        tomeBtn.ZIndex = 19
        tomeBtn.Parent = pFrame
        
        local tomeCorner = Instance.new("UICorner")
        tomeCorner.CornerRadius = UDim.new(0, 6)
        tomeCorner.Parent = tomeBtn
        
        tomeBtn.MouseButton1Click:Connect(function()
            toggleTome(targetPlayer)
        end)
        
        -- Botão TP
        local tpBtn = Instance.new("TextButton")
        tpBtn.Size = UDim2.new(0, 50, 0, 30)
        tpBtn.Position = UDim2.new(1, -60, 0.5, -15)
        tpBtn.BackgroundColor3 = Colors.Primary
        tpBtn.Text = "TP"
        tpBtn.TextColor3 = Colors.Text
        tpBtn.Font = Enum.Font.GothamBold
        tpBtn.TextSize = 12
        tpBtn.ZIndex = 19
        tpBtn.Parent = pFrame
        
        local tpCorner = Instance.new("UICorner")
        tpCorner.CornerRadius = UDim.new(0, 6)
        tpCorner.Parent = tpBtn
        
        tpBtn.MouseButton1Click:Connect(function()
            teleportToPlayer(targetPlayer)
        end)
    end
    
    -- Primeiro os favoritos
    for _, player in ipairs(favoritePlayers) do
        createPlayerFrame(player, true)
    end
    
    -- Depois os não favoritos
    for _, player in ipairs(normalPlayers) do
        createPlayerFrame(player, false)
    end
end
